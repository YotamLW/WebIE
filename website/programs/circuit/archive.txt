    for (let i in this.connections) {
        if (this.connections[i].voltage == this.voltage) { // if voltage is equal the nodes are equal 
            for (let j in this.connections) {
                let removalIndex = this.connections[i].connections.indexOf(this.connection[j]) // index of duplicate connection in other node's connection list (if -1 then no duplicate) 
                if (removalIndex > -1) {this.connections[i].connections.splice(removalIndex, 1)} // removes duplicate connection from other node's list 
            }
            this.connections = this.connections.concat(this.connections[i].connections) // adds the connections of the other node (non-repeating) to the current node

    }
